function removeSmallest(numbers) {
  let newArray = [...numbers];
  let smallestIndex = 0;
  let smallestNumber = Math.min(...numbers);

  for (let = i; i < numbers.length; i++) {
    if (newArray[i] == smallestNumber) {
      smallestIndex = i;
      break;    
    }
  }
  
  newArray.splice(smallestIndex, 1);
  return newArray;
}
